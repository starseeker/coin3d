cmake_minimum_required(VERSION 3.16)

# Add the test to the main Coin3D build (using their build patterns)
configure_file(${CMAKE_CURRENT_SOURCE_DIR}/texture_test_actual.cpp.in ${CMAKE_CURRENT_BINARY_DIR}/texture_test_actual.cpp @ONLY)
add_executable(texture_test_actual ${CMAKE_CURRENT_BINARY_DIR}/texture_test_actual.cpp)
target_link_libraries(texture_test_actual ${PROJECT_NAME} ${COIN_TARGET_LINK_LIBRARIES})

# Add include directories for OSMesa support  
target_include_directories(texture_test_actual PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${PROJECT_SOURCE_DIR}/include
    ${PROJECT_SOURCE_DIR}/include/Inventor/annex
    ${PROJECT_BINARY_DIR}/include
    ${COIN_TARGET_INCLUDE_DIRECTORIES}
)
